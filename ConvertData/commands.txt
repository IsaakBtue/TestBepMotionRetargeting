# === NEW PIPELINE: G1 .pt → MoSh++ (marker positions) → SMPL-X → Booster T1 ===
# Replaces pipeline_g1_pt_to_t1.sh (which used broken analytical coordinate mapping)

# Step 1 — G1 link positions → SMPL-X via MoSh++ (~5–15 min CPU)
conda activate soma && python src/pipeline/g1_pt_to_mosh.py \
    --pt ConvertData/export/unitreeg1/goalkeeper_motion_dataset/lefthand.pt \
    --output-dir output/g1_lefthand
# Add --force to re-run MoSh if output already exists

# Step 1b — SMPL-X → mesh vertices + body animation GIF
python src/pipeline/g1_smplx_to_mesh_pkl.py \
    --input  output/g1_lefthand/smplx_body_mesh_all_frames.pkl \
    --output output/g1_lefthand/smplx_body_mesh_vertices.pkl

conda activate soma
python src/pipeline/create_smplx_gif.py \
    --mesh output/g1_lefthand/smplx_body_mesh_vertices.pkl \
    --out  output/g1_lefthand/smplx_body_animation.gif

# Step 2 — SMPL-X → Booster T1 retargeting
conda activate HumanoidDataGeneration && \
export PYTHONPATH="$PWD/src/retargeting:${PYTHONPATH:-}" && \
python src/retargeting/scripts/mosh_to_robot.py \
    --smplx_mesh_pkl output/g1_lefthand/smplx_body_mesh_all_frames.pkl \
    --robot booster_t1 \
    --mocap_fps 30 \
    --tgt_fps 30 \
    --no_viewer \
    --save_path output/g1_lefthand/retargeting/lefthand_booster.pkl

# Visualize result
./scripts/visualize.sh output/g1_lefthand/retargeting/lefthand_booster.pkl
